## 1. OBJECT CREATION METHODS

### Method 1: Object Literal (‡¶∏‡¶¨‡¶ö‡ßá‡¶Ø‡¶º‡ßá common)

```javascript
const person = {
  name: "John",
  age: 25,
  city: "Dhaka",
  isStudent: true,
};

console.log(person); // { name: "John", age: 25, city: "Dhaka", isStudent: true }
```

### Method 2: new Object()

```javascript
const person = new Object();
person.name = "Jane";
person.age = 30;
person.city = "Chittagong";

console.log(person); // { name: "Jane", age: 30, city: "Chittagong" }
```

### Method 3: Object.create()

```javascript
const personPrototype = {
  greet: function () {
    console.log(`Hello, I'm ${this.name}`);
  },
};

const person = Object.create(personPrototype);
person.name = "Bob";
person.age = 35;

person.greet(); // "Hello, I'm Bob"
```

### Method 4: Constructor Function

```javascript
function Person(name, age, city) {
  this.name = name;
  this.age = age;
  this.city = city;
}

const person1 = new Person("Alice", 28, "Sylhet");
const person2 = new Person("Charlie", 32, "Rajshahi");

console.log(person1); // Person { name: "Alice", age: 28, city: "Sylhet" }
```

### Method 5: Class (ES6+)

```javascript
class Person {
  constructor(name, age, city) {
    this.name = name;
    this.age = age;
    this.city = city;
  }
}

const person = new Person("David", 27, "Khulna");
console.log(person); // Person { name: "David", age: 27, city: "Khulna" }
```

## 2. ACCESSING & MODIFYING PROPERTIES

### Dot Notation vs Bracket Notation

```javascript
const person = {
  name: "John",
  age: 25,
  "favorite color": "blue", // Space ‡¶•‡¶æ‡¶ï‡¶≤‡ßá quotes ‡¶ö‡¶æ‡¶á
  "full-name": "John Doe", // Hyphen ‡¶•‡¶æ‡¶ï‡¶≤‡ßá quotes ‡¶ö‡¶æ‡¶á
};

// ‚úÖ Dot notation
console.log(person.name); // "John"
console.log(person.age); // 25

// ‚úÖ Bracket notation
console.log(person["name"]); // "John"
console.log(person["favorite color"]); // "blue" - dot notation ‡¶ï‡¶æ‡¶ú ‡¶ï‡¶∞‡¶¨‡ßá ‡¶®‡¶æ
console.log(person["full-name"]); // "John Doe"

// ‚ùå These won't work with dot notation
// person.favorite color // SyntaxError
// person.full-name      // NaN (tries to subtract)
```

### Dynamic Property Access

```javascript
const person = {
  name: "John",
  age: 25,
  city: "Dhaka",
};

// Variable ‡¶¶‡¶ø‡¶Ø‡¶º‡ßá property access
const prop = "name";
console.log(person[prop]); // "John"

const key = "age";
console.log(person[key]); // 25

// User input ‡¶•‡ßá‡¶ï‡ßá
function getProperty(obj, propertyName) {
  return obj[propertyName];
}

console.log(getProperty(person, "city")); // "Dhaka"
```

### Adding Properties

```javascript
const person = {
  name: "John",
};

// Dot notation
person.age = 25;
person.city = "Dhaka";

// Bracket notation
person["country"] = "Bangladesh";
person["is-student"] = true;

// Dynamic property name
const key = "email";
person[key] = "john@example.com";

console.log(person);
// {
//   name: "John",
//   age: 25,
//   city: "Dhaka",
//   country: "Bangladesh",
//   "is-student": true,
//   email: "john@example.com"
// }
```

### Modifying Properties

```javascript
const person = {
  name: "John",
  age: 25,
};

// Update
person.name = "Jane";
person["age"] = 30;

console.log(person); // { name: "Jane", age: 30 }
```

### Deleting Properties

```javascript
const person = {
  name: "John",
  age: 25,
  city: "Dhaka",
};

// Delete ‡¶ï‡¶∞‡¶æ
delete person.age;
console.log(person); // { name: "John", city: "Dhaka" }

// Check ‡¶ï‡¶∞‡¶æ property ‡¶Ü‡¶õ‡ßá ‡¶ï‡¶ø‡¶®‡¶æ

console.log("age" in person); // false
```

### Computed Property Names (ES6)

```javascript

const value = "John";

// ‚úÖ Dynamic property name during object creation

  [key]: value,
  [`${key}Length`]: value.length,
  ["age"]: 25,
};

// {
//   name: "John",
//   nameLength: 4,
//   age: 25
// }
```

## 3. OBJECT METHODS

Objects ‡¶è‡¶∞ ‡¶≠‡¶ø‡¶§‡¶∞‡ßá functions ‡¶•‡¶æ‡¶ï‡¶§‡ßá ‡¶™‡¶æ‡¶∞‡ßá - ‡¶∏‡ßá‡¶ó‡ßÅ‡¶≤‡ßã‡¶ï‡ßá methods ‡¶¨‡¶≤‡ßá‡•§

### Basic Methods

```javascript
const person = {
  name: "John",
  age: 25,

  // Method - traditional function
  greet: function () {
    console.log(`Hello, I'm ${this.name}`);
  },

  // Method - ES6 shorthand (‡¶∏‡¶¨‡¶ö‡ßá‡¶Ø‡¶º‡ßá common)
  introduce() {
    console.log(`I'm ${this.name}, ${this.age} years old`);
  },

  // Arrow function (‚ö†Ô∏è this ‡¶ï‡¶æ‡¶ú ‡¶ï‡¶∞‡¶¨‡ßá ‡¶®‡¶æ ‡¶∏‡¶†‡¶ø‡¶ï‡¶≠‡¶æ‡¶¨‡ßá)
  sayAge: () => {
    console.log(this.age); // undefined (this = window/global)
  },
};

person.greet(); // "Hello, I'm John"
person.introduce(); // "I'm John, 25 years old"
person.sayAge(); // undefined
```

### `this` Keyword

```javascript
const calculator = {
  value: 0,

  add(n) {
    this.value += n;
    return this; // Method chaining ‡¶è‡¶∞ ‡¶ú‡¶®‡ßç‡¶Ø
  },

  subtract(n) {
    this.value -= n;
    return this;
  },

  multiply(n) {
    this.value *= n;
    return this;
  },

  getResult() {
    return this.value;
  },
};

// Method chaining
calculator.add(10).multiply(2).subtract(5);
console.log(calculator.getResult()); // 15
```

### Methods with Parameters

```javascript
const bankAccount = {
  balance: 1000,

  deposit(amount) {
    if (amount > 0) {
      this.balance += amount;
      console.log(`Deposited: ‡ß≥${amount}`);
    }
  },

  withdraw(amount) {
    if (amount > 0 && amount <= this.balance) {
      this.balance -= amount;
      console.log(`Withdrawn: ‡ß≥${amount}`);
    } else {
      console.log("Insufficient balance!");
    }
  },

  getBalance() {
    return this.balance;
  },
};

bankAccount.deposit(500); // "Deposited: ‡ß≥500"
bankAccount.withdraw(300); // "Withdrawn: ‡ß≥300"
console.log(bankAccount.getBalance()); // 1200
```

## 5. OBJECT DESTRUCTURING

### Basic Destructuring

```javascript
const person = {
  name: "John",
  age: 25,
  city: "Dhaka",
};

// ‚úÖ Destructuring
const { name, age, city } = person;

console.log(name); // "John"
console.log(age); // 25
console.log(city); // "Dhaka"

// ‚ùå Without destructuring (old way)
const name2 = person.name;
const age2 = person.age;
const city2 = person.city;
```

### Renaming Variables

```javascript
const person = {
  name: "John",
  age: 25,
};

// Variable ‡¶è‡¶∞ name change ‡¶ï‡¶∞‡¶æ
const { name: personName, age: personAge } = person;

console.log(personName); // "John"
console.log(personAge); // 25
// console.log(name);    // ReferenceError
```

### Default Values

```javascript
const person = {
  name: "John",
  age: 25,
};

// Default value ‡¶¶‡ßá‡¶ì‡¶Ø‡¶º‡¶æ (‡¶Ø‡¶¶‡¶ø property ‡¶®‡¶æ ‡¶•‡¶æ‡¶ï‡ßá)
const { name, age, city = "Unknown", country = "Bangladesh" } = person;

console.log(name); // "John"
console.log(age); // 25
console.log(city); // "Unknown" (default)
console.log(country); // "Bangladesh" (default)
```

### Nested Destructuring

```javascript

  name: "John",
  age: 25,
  address: {
    street: "123 Main St",
    city: "Dhaka",
    country: "Bangladesh",
  },
};

// Nested destructuring
const {
  name,
  address: { city, country },
} = user;

console.log(name); // "John"
console.log(city); // "Dhaka"
console.log(country); // "Bangladesh"
// console.log(address); // ReferenceError (address variable ‡¶§‡ßà‡¶∞‡¶ø ‡¶π‡¶Ø‡¶º‡¶®‡¶ø)

// ‡¶Ø‡¶¶‡¶ø address variable ‡¶ì ‡¶ö‡¶æ‡¶á

  name: userName,
  address,
  address: { city: userCity },
} = user;

console.log(address); // { street: "123 Main St", city: "Dhaka", country: "Bangladesh" }
```

### Function Parameter Destructuring

```javascript
// ‚ùå Old way

  console.log(`Name: ${user.name}`);
  console.log(`Age: ${user.age}`);
}

// ‚úÖ With destructuring

  console.log(`Name: ${name}`);
  console.log(`Age: ${age}`);
  console.log(`City: ${city}`);
}

const person = { name: "John", age: 25 };

// Name: John
// Age: 25
// City: Unknown
```

### Rest in Destructuring

```javascript

  name: "John",
  age: 25,
```

## 6. SPREAD & REST WITH OBJECTS

### Spread Operator (...)

#### Copying Objects

```javascript
const original = {
  name: "John",
  age: 25,
};

// ‚úÖ Shallow copy
const copy = { ...original };

console.log(copy); // { name: "John", age: 25 }

// Modify copy
copy.name = "Jane";
console.log(original.name); // "John" (unchanged)
console.log(copy.name); // "Jane"
```

#### Merging Objects

```javascript
const obj1 = { a: 1, b: 2 };
const obj2 = { c: 3, d: 4 };
const obj3 = { e: 5, f: 6 };

// Merge ‡¶ï‡¶∞‡¶æ
const merged = { ...obj1, ...obj2, ...obj3 };
console.log(merged); // { a: 1, b: 2, c: 3, d: 4, e: 5, f: 6 }

// Duplicate keys - ‡¶∂‡ßá‡¶∑‡ßá‡¶∞‡¶ü‡¶æ win ‡¶ï‡¶∞‡ßá
const obj4 = { a: 1, b: 2 };
const obj5 = { b: 3, c: 4 };
const merged2 = { ...obj4, ...obj5 };
console.log(merged2); // { a: 1, b: 3, c: 4 } - obj5 ‡¶è‡¶∞ b win ‡¶ï‡¶∞‡ßá‡¶õ‡ßá
```

#### Adding/Overriding Properties

```javascript
const person = {
  name: "John",
  age: 25,
  city: "Dhaka",
};

// ‡¶®‡¶§‡ßÅ‡¶® property add ‡¶ï‡¶∞‡¶æ
const updatedPerson = {
  ...person,
  email: "john@example.com",
  phone: "01712345678",
};

console.log(updatedPerson);
// {
//   name: "John",
//   age: 25,
//   city: "Dhaka",
//   email: "john@example.com",
//   phone: "01712345678"
// }

// Property override ‡¶ï‡¶∞‡¶æ
const olderPerson = {
  ...person,
  age: 30, // Override
};

console.log(olderPerson);
// { name: "John", age: 30, city: "Dhaka" }
```

#### Removing Properties

```javascript
const user = {
  name: "John",
  age: 25,
  password: "secret123",
  email: "john@example.com",
};

// password ‡¶¨‡¶æ‡¶¶‡ßá ‡¶¨‡¶æ‡¶ï‡¶ø ‡¶∏‡¶¨
const { password, ...userWithoutPassword } = user;

console.log(userWithoutPassword);
// {
//   name: "John",
//   age: 25,
//   email: "john@example.com"
// }
```

## 7. OBJECT BUILT-IN METHODS

### Object.keys(), Object.values(), Object.entries()

Already covered above, here's a summary:

```javascript
const person = { name: "John", age: 25, city: "Dhaka" };

Object.keys(person); // ["name", "age", "city"]
```

### Object.assign()

```javascript
// Copy ‡¶ï‡¶∞‡¶æ

// Merge ‡¶ï‡¶∞‡¶æ
const obj1 = { a: 1, b: 2 };
const obj2 = { c: 3 };
const obj3 = { d: 4 };

// Target object modify ‡¶π‡¶Ø‡¶º

console.log(target); // { a: 1, b: 2 } - modified!

// ‚ö†Ô∏è Spread operator use ‡¶ï‡¶∞‡¶æ better
const merged2 = { ...obj1, ...obj2, ...obj3 };
```

### Object.freeze()

```javascript

  name: "John",
  age: 25,
};

// Freeze ‡¶ï‡¶∞‡¶æ - ‡¶ï‡ßã‡¶®‡ßã changes allowed ‡¶®‡¶æ

person.age = 30; // Silent fail (strict mode ‡¶è error)
person.city = "Dhaka"; // Silent fail
delete person.name; // Silent fail

console.log(person); // { name: "John", age: 25 } - unchanged

// Check ‡¶ï‡¶∞‡¶æ frozen ‡¶ï‡¶ø‡¶®‡¶æ
```

### Object.seal()

```javascript
const person = {
  name: "John",
  age: 25,
};

// Seal ‡¶ï‡¶∞‡¶æ - modify ‡¶ï‡¶∞‡¶æ ‡¶Ø‡¶æ‡¶¨‡ßá, add/delete ‡¶ï‡¶∞‡¶æ ‡¶Ø‡¶æ‡¶¨‡ßá ‡¶®‡¶æ
Object.seal(person);

person.age = 30; // ‚úÖ Works - modify allowed
person.city = "Dhaka"; // ‚ùå Fails - add not allowed
delete person.name; // ‚ùå Fails - delete not allowed

// Check ‡¶ï‡¶∞‡¶æ sealed ‡¶ï‡¶ø‡¶®‡¶æ

```

### Freeze vs Seal

```javascript
const frozen = Object.freeze({ a: 1, b: 2 });
const sealed = Object.seal({ a: 1, b: 2 });

// Modify

sealed.a = 10; // ‚úÖ Allowed

// Add
frozen.c = 3; // ‚ùå Not allowed
sealed.c = 3; // ‚ùå Not allowed

// Delete
delete frozen.a; // ‚ùå Not allowed
delete sealed.a; // ‚ùå Not allowed

console.log(frozen); // { a: 1, b: 2 }
console.log(sealed); // { a: 10, b: 2 }
```

### Object.fromEntries()


### Object.hasOwnProperty() & in operator

## 8. OBJECT REFERENCES & CLONING

### Reference vs Value

```javascript
// Primitives - value copy ‡¶π‡¶Ø‡¶º
let a = 10;
let b = a;
b = 20;
console.log(a); // 10 (unchanged)

// Objects - reference copy ‡¶π‡¶Ø‡¶º
let obj1 = { name: "John" };
let obj2 = obj1; // Same reference!

obj2.name = "Jane";
console.log(obj1.name); // "Jane" (changed!) ‚ö†Ô∏è
console.log(obj1 === obj2); // true (same reference)
```

### Shallow Copy

```javascript
const original = {
  name: "John",
  age: 25,
  address: {
    city: "Dhaka",
  },
};

// Method 1: Spread operator
const copy1 = { ...original };

// Method 2: Object.assign()
const copy2 = Object.assign({}, original);

// Shallow copy - nested objects still referenced
copy1.name = "Jane"; // ‚úÖ Independent
copy1.address.city = "Chittagong"; // ‚ö†Ô∏è Affects original!

console.log(original.name); // "John" (unchanged)
console.log(original.address.city); // "Chittagong" (changed!)
```

### Deep Copy

```javascript
const original = {
  name: "John",
  age: 25,
  address: {
    city: "Dhaka",
    country: "Bangladesh",
  },
  hobbies: ["reading", "coding"],
};

// ‚úÖ Method 1: JSON (limitations ‡¶Ü‡¶õ‡ßá)
const deepCopy1 = JSON.parse(JSON.stringify(original));

deepCopy1.address.city = "Chittagong";
console.log(original.address.city); // "Dhaka" (unchanged)

// ‚ö†Ô∏è JSON method ‡¶è‡¶∞ limitations:
// - Functions copy ‡¶π‡¶Ø‡¶º ‡¶®‡¶æ
// - undefined values ‡¶π‡¶æ‡¶∞‡¶ø‡¶Ø‡¶º‡ßá ‡¶Ø‡¶æ‡¶Ø‡¶º
// - Dates string ‡¶è convert ‡¶π‡¶Ø‡¶º
// - Circular references handle ‡¶ï‡¶∞‡¶§‡ßá ‡¶™‡¶æ‡¶∞‡ßá ‡¶®‡¶æ

const obj = {
  func: function () {}, // ‡¶π‡¶æ‡¶∞‡¶ø‡¶Ø‡¶º‡ßá ‡¶Ø‡¶æ‡¶¨‡ßá
  undef: undefined, // ‡¶π‡¶æ‡¶∞‡¶ø‡¶Ø‡¶º‡ßá ‡¶Ø‡¶æ‡¶¨‡ßá
  date: new Date(), // String ‡¶π‡¶Ø‡¶º‡ßá ‡¶Ø‡¶æ‡¶¨‡ßá
};

const copy = JSON.parse(JSON.stringify(obj));
console.log(copy); // { date: "2025-..." } - func ‡¶è‡¶¨‡¶Ç undef ‡¶®‡¶æ‡¶á!

// ‚úÖ Method 2: structuredClone() (Modern - ES2022)
```

```javascript
          current: this.current[key],
          default: this.defaults[key],
        };
      }
    }
```

### Object Equality Check (Deep)

```javascript
function deepEqual(obj1, obj2) {
  // Same reference
  if (obj1 === obj2) return true;

  // Check types
  if (
    typeof obj1 !== "object" ||
    typeof obj2 !== "object" ||
    obj1 === null ||
    obj2 === null
  ) {
    return false;
  }

  // Get keys
  const keys1 = Object.keys(obj1);
  const keys2 = Object.keys(obj2);

  // Different number of properties
  if (keys1.length !== keys2.length) {
    return false;
  }

  // Check each property recursively
  for (let key of keys1) {
    if (!keys2.includes(key)) {
      return false;
    }

    if (!deepEqual(obj1[key], obj2[key])) {
      return false;
    }
  }

  return true;
}
```

### Object Pool Pattern

```javascript
const objectPool = {
  pool: [],
  maxSize: 10,

  get() {
    if (this.pool.length > 0) {
      console.log("Reusing object from pool");
      return this.pool.pop();
    }

    console.log("Creating new object");
    return {};
  },

  release(obj) {

      for (let key in obj) {
        delete obj[key];
      }
      this.pool.push(obj);

    }
  },

  getPoolSize() {
    return this.pool.length;
  },
};

obj2.name = "Jane";

objectPool.release(obj2); // Object returned to pool
```

```javascript
// 2. Don't modify objects passed as parameters (unless intended)
```
