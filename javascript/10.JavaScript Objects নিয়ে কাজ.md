## 8. OBJECT REFERENCES & CLONING

### Reference vs Value

```javascript

let b = a;
b = 20;
console.log(a); // 10 (unchanged)

// Objects - reference copy à¦¹à¦¯à¦¼

let obj2 = obj1; // Same reference!

obj2.name = "Jane";
console.log(obj1.name); // "Jane" (changed!) âš ï¸
console.log(obj1 === obj2); // true (same reference)
```

### Shallow Copy

```javascript

  name: "John",
  age: 25,
  address: {
    city: "Dhaka",
  },
};

// Method 1: Spread operator
const copy1 = { ...original };

// Method 2: Object.assign()
const copy2 = Object.assign({}, original);

// Shallow copy - nested objects still referenced
copy1.name = "Jane"; // âœ… Independent
copy1.address.city = "Chittagong"; // âš ï¸ Affects original!

console.log(original.name); // "John" (unchanged)
console.log(original.address.city); // "Chittagong" (changed!)
```

### Deep Copy

```javascript
const original = {
  name: "John",
  age: 25,
  address: {
    city: "Dhaka",
    country: "Bangladesh",
  },
  hobbies: ["reading", "coding"],
};

// âœ… Method 1: JSON (limitations à¦†à¦›à§‡)
const deepCopy1 = JSON.parse(JSON.stringify(original));

deepCopy1.address.city = "Chittagong";
console.log(original.address.city); // "Dhaka" (unchanged)

// âš ï¸ JSON method à¦à¦° limitations:
// - Functions copy à¦¹à¦¯à¦¼ à¦¨à¦¾
// - undefined values à¦¹à¦¾à¦°à¦¿à¦¯à¦¼à§‡ à¦¯à¦¾à¦¯à¦¼
// - Dates string à¦ convert à¦¹à¦¯à¦¼
// - Circular references handle à¦•à¦°à¦¤à§‡ à¦ªà¦¾à¦°à§‡ à¦¨à¦¾

const obj = {
  func: function () {}, // à¦¹à¦¾à¦°à¦¿à¦¯à¦¼à§‡ à¦¯à¦¾à¦¬à§‡
  undef: undefined, // à¦¹à¦¾à¦°à¦¿à¦¯à¦¼à§‡ à¦¯à¦¾à¦¬à§‡
  date: new Date(), // String à¦¹à¦¯à¦¼à§‡ à¦¯à¦¾à¦¬à§‡
};

const copy = JSON.parse(JSON.stringify(obj));
console.log(copy); // { date: "2025-..." } - func à¦à¦¬à¦‚ undef à¦¨à¦¾à¦‡!

// âœ… Method 2: structuredClone() (Modern - ES2022)
const deepCopy2 = structuredClone(original);

deepCopy2.address.city = "Sylhet";
console.log(original.address.city); // "Dhaka" (unchanged)

// âœ… Method 3: Custom deep clone function
function deepClone(obj) {
  if (obj === null || typeof obj !== "object") {
    return obj;
  }

  if (obj instanceof Date) {
    return new Date(obj);
  }

  if (obj instanceof Array) {
    return obj.map((item) => deepClone(item));
  }

  const cloned = {};
  for (let key in obj) {
    if (obj.hasOwnProperty(key)) {
      cloned[key] = deepClone(obj[key]);
    }
  }

  return cloned;
}

const deepCopy3 = deepClone(original);
```

### Object Equality Check (Deep)

```javascript
function deepEqual(obj1, obj2) {
  // Same reference
  if (obj1 === obj2) return true;

  // Check types
  if (
    typeof obj1 !== "object" ||
    typeof obj2 !== "object" ||
    obj1 === null ||
    obj2 === null
  ) {
    return false;
  }

  // Get keys
  const keys1 = Object.keys(obj1);
  const keys2 = Object.keys(obj2);

  // Different number of properties
  if (keys1.length !== keys2.length) {
    return false;
  }

  // Check each property recursively
  for (let key of keys1) {
    if (!keys2.includes(key)) {
      return false;
    }

    if (!deepEqual(obj1[key], obj2[key])) {
      return false;
    }
  }

  return true;
}
```

### Object Pool Pattern

```javascript
const objectPool = {
  pool: [],
  maxSize: 10,

  get() {
    if (this.pool.length > 0) {
      console.log("Reusing object from pool");
      return this.pool.pop();
    }

    console.log("Creating new object");
    return {};
  },

  release(obj) {
    if (this.pool.length < this.maxSize) {
      // Reset object
      for (let key in obj) {
        delete obj[key];
      }
      this.pool.push(obj);
      console.log("Object returned to pool");
    }
  },

  getPoolSize() {
    return this.pool.length;
  },
};

obj2.name = "Jane";

objectPool.release(obj1); // Object returned to pool
objectPool.release(obj2); // Object returned to pool

const obj3 = objectPool.get(); // Reusing object from pool
console.log(obj3); // {}
```
